import numpy as np
import matplotlib.pyplot as plt
%matplotlib inline

from sklearn import datasets
from sklearn.decomposition import PCA

iris = datasets.load_iris()

x = iris.data
y = iris.target

x.shape

(150, 4)

pca = PCA(n_components=2)

pca.fit(x)
